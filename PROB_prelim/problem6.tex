\begin{exercise}
Which set(s) of parametric equations describe the line containing $P(3, -2, 7)$, if the line is perpendicular to the plane $2x+y-5z=10$? 

Select all that apply.

\begin{selectAll}
    \choice[correct]{$x=2t+3$; $y=t-2$; $z=-5t+7$.}
   \choice{$x=3t+2$; $y=-2t+1$; $z=7t-5$.}
   \choice{$x=10t+3$; $y=10t-2$; $z=10t+7$.}
   \choice[correct]{$x=-2t+3$; $y=-t-2$; $z=5t+7$.}
   \end{selectAll}
 
\end{exercise}
